Task 14.2 — Meaningful Names, Single Responsibility & No Magic Numbers

Exercise 1: Meaningful Names

❌ BAD Names

const d = new Date();
const x = users.filter(u => u.a > 18);
function calc(a, b) {
  return a * b * 0.1;
}

✅ GOOD Names

const currentDate = new Date();
const adultUsers = users.filter(
  user => user.age > 18
);
function calculateDiscount(price, qty) {
  const DISCOUNT_RATE = 0.1;
  return price * qty * DISCOUNT_RATE;
}

Exercise 2: Single Responsibility

❌ BAD — Does too many things

function processUser(userData) {
  // Validate
  // Transform
  // Save to DB
  // Send email
  // Update UI
  // TOO MANY JOBS!
}

✅ GOOD — One job each

validateUser(userData);
normalizeUser(userData);
createUser(userData);
notifyUser(userData);
renderUser(userData);
