Task 14.3 — Console Methods, Breakpoints & Bug Fixing

Exercise 1: Console Methods

Open DevTools (F12) → Console to see all outputs

console.log()

Basic logging for any value

console.log("Basic message")

console.warn()

Yellow warning messages

console.warn("This might be a problem")

console.error()

Red error messages

console.error("This is wrong")

console.table()

Displays arrays/objects as table

console.table(users)

console.group()

Groups related logs together

console.group("User Processing")

console.time()

Measures how long code takes

console.time("fetchUsers")

console.assert()

Logs only if condition is false

console.assert(x > 0, "x must be positive")

console.trace()

Shows call stack at that point

console.trace("How did we get here?")

✅ Check your DevTools Console (F12) to see all outputs!

Exercise 3: Debug This Code

❌ BUGGY Code

function calculateOrderTotal(items) {
  let total = 0;

  // BUG 1: i <= items.length
  // causes undefined on last iteration
  for (let i = 0; i <= items.length; i++) {
    const item = items[i];
    // BUG 2: typo - item.quanity
    total += item.price * item.quanity;
  }

  // BUG 3: wrong threshold
  // should be >= not >
  if (total > 100) {
    total = total * 0.9;
  }

  return total;
}

✅ FIXED Code

function calculateOrderTotal(items) {
  let total = 0;

  // FIX 1: i < items.length (not <=)
  for (let i = 0; i < items.length; i++) {
    const item = items[i];
    // FIX 2: correct spelling - quantity
    total += item.price * item.quantity;
  }

  // FIX 3: >= 100 for exact threshold
  if (total >= 100) {
    total = total * 0.9;
  }

  return total;
}
